191  constexpr double matthews(const ConfusionMatrixBase<T>& matrix) {
192  return std::sqrt(true_positive_rate(matrix) * true_negative_rate(matrix) * positive_predictive_value(matrix) *
193  negative_predictive_value(matrix)) - std::sqrt(false_negative_rate(matrix) *

202  template<class T>
203  constexpr double f_beta(const ConfusionMatrixBase<T>& matrix, double beta) {
204  double bs = beta * beta;
205  double num = (1.0 + bs) * static_cast<double>(matrix.TruePositives);
206  double den = num + bs * static_cast<double>(matrix.FalseNegatives) + static_cast<double>(matrix.FalsePositives);
207  if(den == 0.0) return 0.0;
208  return num / den;
209  }
